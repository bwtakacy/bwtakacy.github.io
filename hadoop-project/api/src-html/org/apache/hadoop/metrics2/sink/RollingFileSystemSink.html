051/**
052 * <p>This class is a metrics sink that uses
053 * {@link org.apache.hadoop.fs.FileSystem} to write the metrics logs.  Every
054 * hour a new directory will be created under the path specified by the
055 * <code>basepath</code> property. All metrics will be logged to a file in the
056 * current hour's directory in a file named &lt;hostname&gt;.log, where
057 * &lt;hostname&gt; is the name of the host on which the metrics logging
058 * process is running. The base path is set by the
059 * <code>&lt;prefix&gt;.sink.&lt;instance&gt;.basepath</code> property.  The
060 * time zone used to create the current hour's directory name is GMT.  If the
061 * <code>basepath</code> property isn't specified, it will default to
062 * &quot;/tmp&quot;, which is the temp directory on whatever default file
063 * system is configured for the cluster.</p>
064 *
065 * <p>The <code>&lt;prefix&gt;.sink.&lt;instance&gt;.ignore-error</code>
066 * property controls whether an exception is thrown when an error is encountered
067 * writing a log file.  The default value is <code>true</code>.  When set to
068 * <code>false</code>, file errors are quietly swallowed.</p>
069 *
070 * <p>The primary use of this class is for logging to HDFS.  As it uses
071 * {@link org.apache.hadoop.fs.FileSystem} to access the target file system,
072 * however, it can be used to write to the local file system, Amazon S3, or any
073 * other supported file system.  The base path for the sink will determine the
074 * file system used.  An unqualified path will write to the default file system
075 * set by the configuration.</p>
076 *
077 * <p>Not all file systems support the ability to append to files.  In file
078 * systems without the ability to append to files, only one writer can write to
079 * a file at a time.  To allow for concurrent writes from multiple daemons on a
080 * single host, the <code>source</code> property should be set to the name of
081 * the source daemon, e.g. <i>namenode</i>.  The value of the
082 * <code>source</code> property should typically be the same as the property's
083 * prefix.  If this property is not set, the source is taken to be
084 * <i>unknown</i>.</p>
085 *
086 * <p>Instead of appending to an existing file, by default the sink
087 * will create a new file with a suffix of &quot;.&lt;n&gt;&quet;, where
088 * <i>n</i> is the next lowest integer that isn't already used in a file name,
089 * similar to the Hadoop daemon logs.  NOTE: the file with the <b>highest</b>
090 * sequence number is the <b>newest</b> file, unlike the Hadoop daemon logs.</p>
091 *
092 * <p>For file systems that allow append, the sink supports appending to the
093 * existing file instead. If the <code>allow-append</code> property is set to
094 * true, the sink will instead append to the existing file on file systems that
095 * support appends. By default, the <code>allow-append</code> property is
096 * false.</p>
097 *
098 * <p>Note that when writing to HDFS with <code>allow-append</code> set to true,
099 * there is a minimum acceptable number of data nodes.  If the number of data
100 * nodes drops below that minimum, the append will succeed, but reading the
101 * data will fail with an IOException in the DataStreamer class.  The minimum
102 * number of data nodes required for a successful append is generally 2 or
103 * 3.</p>
104 *
105 * <p>Note also that when writing to HDFS, the file size information is not
106 * updated until the file is closed (e.g. at the top of the hour) even though
107 * the data is being written successfully. This is a known HDFS limitation that
108 * exists because of the performance cost of updating the metadata.  See
109 * <a href="https://issues.apache.org/jira/browse/HDFS-5478">HDFS-5478</a>.</p>
110 */
111@InterfaceAudience.Public
112@InterfaceStability.Evolving
113public class RollingFileSystemSink implements MetricsSink, Closeable {
114  private static final String BASEPATH_KEY = "basepath";
115  private static final String SOURCE_KEY = "source";
116  private static final String IGNORE_ERROR_KEY = "ignore-error";
117  private static final String ALLOW_APPEND_KEY = "allow-append";
118  private static final String SOURCE_DEFAULT = "unknown";
119  private static final String BASEPATH_DEFAULT = "/tmp";
120  private static final FastDateFormat DATE_FORMAT =
121      FastDateFormat.getInstance("yyyyMMddHH", TimeZone.getTimeZone("GMT"));
122  private final Object lock = new Object();
123  private String source;
124  private boolean ignoreError;
125  private boolean allowAppend;
126  private Path basePath;
127  private FileSystem fileSystem;
128  // The current directory path into which we're writing files
129  private Path currentDirPath;
130  // The path to the current file into which we're writing data
131  private Path currentFilePath;
132  // The stream to which we're currently writing.
133  private PrintStream currentOutStream;
134  // We keep this only to be able to call hsynch() on it.
135  private FSDataOutputStream currentFSOutStream;
136  private Timer flushTimer;
137  @VisibleForTesting
138  protected static boolean isTest = false;
139  @VisibleForTesting
140  protected static volatile boolean hasFlushed = false;
141
142  @Override
143  public void init(SubsetConfiguration conf) {
144    basePath = new Path(conf.getString(BASEPATH_KEY, BASEPATH_DEFAULT));
145    source = conf.getString(SOURCE_KEY, SOURCE_DEFAULT);
146    ignoreError = conf.getBoolean(IGNORE_ERROR_KEY, false);
147    allowAppend = conf.getBoolean(ALLOW_APPEND_KEY, false);
148
149    try {
150      fileSystem = FileSystem.get(new URI(basePath.toString()),
151          new Configuration());
152    } catch (URISyntaxException ex) {
153      throw new MetricsException("The supplied filesystem base path URI"
154          + " is not a valid URI: " + basePath.toString(), ex);
155    } catch (IOException ex) {
156      throw new MetricsException("Error connecting to file system: "
157          + basePath + " [" + ex.toString() + "]", ex);
158    }
159
160    // If we're permitted to append, check if we actually can
161    if (allowAppend) {
162      allowAppend = checkAppend(fileSystem);
163    }
164
165    flushTimer = new Timer("RollingFileSystemSink Flusher", true);
166  }
167
168  /**
169   * Test whether the file system supports append and return the answer.
170   * @param fs the target file system
171   */
172  private boolean checkAppend(FileSystem fs) {
173    boolean canAppend = true;
174
175    try {
176      fs.append(basePath);
177    } catch (IOException ex) {
178      if (ex.getMessage().equals("Not supported")) {
179        canAppend = false;
180      }
181    }
182
183    return canAppend;
184  }
185
186  /**
187   * Check the current directory against the time stamp.  If they're not
188   * the same, create a new directory and a new log file in that directory.
189   *
190   * @throws MetricsException thrown if an error occurs while creating the
191   * new directory or new log file
192   */
193  private void rollLogDirIfNeeded() throws MetricsException {
194    Date now = new Date();
195    String currentDir = DATE_FORMAT.format(now);
196    Path path = new Path(basePath, currentDir);
197
198    // We check whether currentOutStream is null instead of currentDirPath,
199    // because if currentDirPath is null, then currentOutStream is null, but
200    // currentOutStream can be null for other reasons.
201    if ((currentOutStream == null) || !path.equals(currentDirPath)) {
202      // Close the stream. This step could have been handled already by the
203      // flusher thread, but if it has, the PrintStream will just swallow the
204      // exception, which is fine.
205      if (currentOutStream != null) {
206        currentOutStream.close();
207      }
208
209      currentDirPath = path;
210
211      try {
212        rollLogDir();
213      } catch (IOException ex) {
214        throwMetricsException("Failed to create new log file", ex);
215      }
216
217      scheduleFlush(now);
218    }
219  }
220
221  /**
222   * Schedule the current hour's directory to be flushed at the top of the next
223   * hour. If this ends up running after the top of the next hour, it will
224   * execute immediately.
225   *
226   * @param now the current time
227   */
228  private void scheduleFlush(Date now) {
229    // Store the current currentDirPath to close later
230    final PrintStream toClose = currentOutStream;
231    Calendar next = Calendar.getInstance();
232
233    next.setTime(now);
234
235    if (isTest) {
236      // If we're running unit tests, flush after a short pause
237      next.add(Calendar.MILLISECOND, 400);
238    } else {
239      // Otherwise flush at the top of the hour
240      next.set(Calendar.SECOND, 0);
241      next.set(Calendar.MINUTE, 0);
242      next.add(Calendar.HOUR, 1);
243    }
244
245    flushTimer.schedule(new TimerTask() {
246      @Override
247      public void run() {
248        synchronized (lock) {
249          // This close may have already been done by a putMetrics() call. If it
250          // has, the PrintStream will swallow the exception, which is fine.
251          toClose.close();
252        }
253
254        hasFlushed = true;
255      }
256    }, next.getTime());
257  }
258
259  /**
260   * Create a new directory based on the current hour and a new log file in
261   * that directory.
262   *
263   * @throws IOException thrown if an error occurs while creating the
264   * new directory or new log file
265   */
266  private void rollLogDir() throws IOException {
267    String fileName =
268        source + "-" + InetAddress.getLocalHost().getHostName() + ".log";
269
270    Path targetFile = new Path(currentDirPath, fileName);
271    fileSystem.mkdirs(currentDirPath);
272
273    if (allowAppend) {
274      createOrAppendLogFile(targetFile);
275    } else {
276      createLogFile(targetFile);
277    }
278  }
279
280  /**
281   * Create a new log file and return the {@link FSDataOutputStream}. If a
282   * file with the specified path already exists, add a suffix, starting with 1
283   * and try again. Keep incrementing the suffix until a nonexistent target
284   * path is found.
285   *
286   * Once the file is open, update {@link #currentFSOutStream},
287   * {@link #currentOutStream}, and {@#link #currentFile} are set appropriately.
288   *
289   * @param initial the target path
290   * @throws IOException thrown if the call to see if the exists fails
291   */
292  private void createLogFile(Path initial) throws IOException {
293    Path currentAttempt = initial;
294    // Start at 0 so that if the base filname exists, we start with the suffix
295    // ".1".
296    int id = 0;
297
298    while (true) {
299      // First try blindly creating the file. If we fail, it either means
300      // the file exists, or the operation actually failed.  We do it this way
301      // because if we check whether the file exists, it might still be created
302      // by the time we try to create it. Creating first works like a
303      // test-and-set.
304      try {
305        currentFSOutStream = fileSystem.create(currentAttempt, false);
306        currentOutStream = new PrintStream(currentFSOutStream, true,
307            StandardCharsets.UTF_8.name());
308        currentFilePath = currentAttempt;
309        break;
310      } catch (IOException ex) {
311        // Now we can check to see if the file exists to know why we failed
312        if (fileSystem.exists(currentAttempt)) {
313          id = getNextIdToTry(initial, id);
314          currentAttempt = new Path(initial.toString() + "." + id);
315        } else {
316          throw ex;
317        }
318      }
319    }
320  }
321
322  /**
323   * Return the next ID suffix to use when creating the log file. This method
324   * will look at the files in the directory, find the one with the highest
325   * ID suffix, and 1 to that suffix, and return it. This approach saves a full
326   * linear probe, which matters in the case where there are a large number of
327   * log files.
328   *
329   * @param initial the base file path
330   * @param lastId the last ID value that was used
331   * @return the next ID to try
332   * @throws IOException thrown if there's an issue querying the files in the
333   * directory
334   */
335  private int getNextIdToTry(Path initial, int lastId)
336      throws IOException {
337    RemoteIterator<LocatedFileStatus> files =
338        fileSystem.listFiles(currentDirPath, true);
339    String base = initial.toString();
340    int id = lastId;
341
342    while (files.hasNext()) {
343      String file = files.next().getPath().getName();
344
345      if (file.startsWith(base)) {
346        int fileId = extractId(file);
347
348        if (fileId > id) {
349          id = fileId;
350        }
351      }
352    }
353
354    // Return either 1 more than the highest we found or 1 more than the last
355    // ID used (if no ID was found).
356    return id + 1;
357  }
358
359  /**
360   * Extract the ID from the suffix of the given file name.
361   *
362   * @param file the file name
363   * @return the ID or -1 if no ID could be extracted
364   */
365  private int extractId(String file) {
366    int index = file.lastIndexOf(".");
367    int id = -1;
368
369    // A hostname has to have at least 1 character
370    if (index > 0) {
371      try {
372        id = Integer.parseInt(file.substring(index + 1));
373      } catch (NumberFormatException ex) {
374        // This can happen if there's no suffix, but there is a dot in the
375        // hostname.  Just ignore it.
376      }
377    }
378
379    return id;
380  }

382  /**
383   * Create a new log file and return the {@link FSDataOutputStream}. If a
384   * file with the specified path already exists, open the file for append
385   * instead.
386   *
387   * Once the file is open, update {@link #currentFSOutStream},
388   * {@link #currentOutStream}, and {@#link #currentFile} are set appropriately.
389   *
390   * @param initial the target path
391   * @throws IOException thrown if the call to see the append operation fails.
392   */
393  private void createOrAppendLogFile(Path targetFile) throws IOException {
394    // First try blindly creating the file. If we fail, it either means
395    // the file exists, or the operation actually failed.  We do it this way
396    // because if we check whether the file exists, it might still be created
397    // by the time we try to create it. Creating first works like a
398    // test-and-set.
399    try {
400      currentFSOutStream = fileSystem.create(targetFile, false);
401      currentOutStream = new PrintStream(currentFSOutStream, true,
402          StandardCharsets.UTF_8.name());
403    } catch (IOException ex) {
404      // Try appending instead.  If we fail, if means the file doesn't
405      // actually exist yet or the operation actually failed.
406      try {
407        currentFSOutStream = fileSystem.append(targetFile);
408        currentOutStream = new PrintStream(currentFSOutStream, true,
409            StandardCharsets.UTF_8.name());
410      } catch (IOException ex2) {
411        // If the original create failed for a legit but transitory
412        // reason, the append will fail because the file now doesn't exist,
413        // resulting in a confusing stack trace.  To avoid that, we set
414        // the cause of the second exception to be the first exception.
415        // It's still a tiny bit confusing, but it's enough
416        // information that someone should be able to figure it out.
417        ex2.initCause(ex);
418
419        throw ex2;
420      }
421    }
422
423    currentFilePath = targetFile;
424  }

426  @Override
427  public void putMetrics(MetricsRecord record) {
428    synchronized (lock) {
429      rollLogDirIfNeeded();
430
431      if (currentOutStream != null) {
432        currentOutStream.printf("%d %s.%s", record.timestamp(),
433            record.context(), record.name());
434
435        String separator = ": ";
436
437        for (MetricsTag tag : record.tags()) {
438          currentOutStream.printf("%s%s=%s", separator, tag.name(),
439              tag.value());
440          separator = ", ";
441        }
442
443        for (AbstractMetric metric : record.metrics()) {
444          currentOutStream.printf("%s%s=%s", separator, metric.name(),
445              metric.value());
446        }
447
448        currentOutStream.println();
449
450        // If we don't hflush(), the data may not be written until the file is
451        // closed. The file won't be closed until the top of the hour *AND*
452        // another record is received. Calling hflush() makes sure that the data
453        // is complete at the top of the hour.
454        try {
455          currentFSOutStream.hflush();
456        } catch (IOException ex) {
457          throwMetricsException("Failed flushing the stream", ex);
458        }
459
460        checkForErrors("Unable to write to log file");
461      } else if (!ignoreError) {
462        throwMetricsException("Unable to write to log file");
463      }
464    }
465  }
466
467  @Override
468  public void flush() {
469    synchronized (lock) {
470      // currentOutStream is null if currentFSOutStream is null
471      if (currentFSOutStream != null) {
472        try {
473          currentFSOutStream.hflush();
474        } catch (IOException ex) {
475          throwMetricsException("Unable to flush log file", ex);
476        }
477      }
478    }
479  }
480
481  @Override
482  public void close() {
483    synchronized (lock) {
484      if (currentOutStream != null) {
485        currentOutStream.close();
486
487        try {
488          checkForErrors("Unable to close log file");
489        } finally {
490          // Null out the streams just in case someone tries to reuse us.
491          currentOutStream = null;
492          currentFSOutStream = null;
493        }
494      }
495    }
496  }

498  /**
499   * If the sink isn't set to ignore errors, throw a {@link MetricsException}
500   * if the stream encountered an exception.  The message parameter will be used
501   * as the new exception's message with the current file name
502   * ({@link #currentFilePath}) appended to it.
503   *
504   * @param message the exception message. The message will have the current
505   * file name ({@link #currentFilePath}) appended to it.
506   * @throws MetricsException thrown if there was an error and the sink isn't
507   * ignoring errors
508   */
509  private void checkForErrors(String message)
510      throws MetricsException {
511    if (!ignoreError && currentOutStream.checkError()) {
512      throw new MetricsException(message + ": " + currentFilePath);
513    }
514  }
515
516  /**
517   * If the sink isn't set to ignore errors, wrap the Throwable in a
518   * {@link MetricsException} and throw it.  The message parameter will be used
519   * as the new exception's message with the current file name
520   * ({@link #currentFilePath}) and the Throwable's string representation
521   * appended to it.
522   *
523   * @param message the exception message. The message will have the current
524   * file name ({@link #currentFilePath}) and the Throwable's string
525   * representation appended to it.
526   * @param t the Throwable to wrap
527   */
528  private void throwMetricsException(String message, Throwable t) {
529    if (!ignoreError) {
530      throw new MetricsException(message + ": " + currentFilePath + " ["
531          + t.toString() + "]", t);
532    }
533  }
534
535  /**
536   * If the sink isn't set to ignore errors, throw a new
537   * {@link MetricsException}.  The message parameter will be used  as the
538   * new exception's message with the current file name
539   * ({@link #currentFilePath}) appended to it.
540   *
541   * @param message the exception message. The message will have the current
542   * file name ({@link #currentFilePath}) appended to it.
543   */
544  private void throwMetricsException(String message) {
545    if (!ignoreError) {
546      throw new MetricsException(message + ": " + currentFilePath);
547    }
548  }
